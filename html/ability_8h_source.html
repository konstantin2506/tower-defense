158 void tryUseAbility([[maybe_unused]]float dt, IEnemy& owner) override
159 {
160 timer_.update(dt);
161 if (timer_.isPassed()){
162
163 std::vector<ID> clients = clientService.getInRadius(owner.getPosition(), radius_);
164
165 std::ranges::for_each(clients, [&](auto id)
166 {
167 if (id != owner.getID()) {
168 clientService.addEffect(id, effect_->clone());
169 }
170 });
171 timer_.reset();
172 }
173 }

183 ability->effect_ = std::unique_ptr<EffectT>(static_cast<EffectT*>(cloned_effect.release()));
