186 std::vector<T> getInRadius(const Matrix<T>& matrix, Vector2<size_t> point, size_t radius){
187 std::vector<T> result;
188 size_t startX = std::max(static_cast<int>(point.getX()) - static_cast<int>(radius), 0);
189 size_t startY = std::max(static_cast<int>(point.getY()) - static_cast<int>(radius), 0);
190 size_t endX = std::min(point.getX() + radius, matrix.rows() - 1);
191 size_t endY = std::min(point.getY() + radius, matrix.cols() - 1);
192 auto rq = radius * radius;
193
194 for (auto x = startX; x <= endX; ++x) {
195 for (auto y = startY; y <= endY; ++y) {
196 if (Vector2<size_t>::distanceQ(point, Vector2{x, y}) <= rq) {
197 result.emplace_back(matrix[x][y]);
198 }
199 }
200 }
201 return result;
202 }
